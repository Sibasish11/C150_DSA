# Write a program to implement a Priority Queue using an array (simple unsorted array where dequeue finds min/max).

```c

#include <stdio.h>
#include <stdlib.h>

#define MAX 100

typedef struct {
    int arr[MAX];
    int size;
} PriorityQueue;

// Initialize Priority Queue
void init(PriorityQueue* pq) {
    pq->size = 0;
}

// Check if empty
int isEmpty(PriorityQueue* pq) {
    return (pq->size == 0);
}

// Check if full
int isFull(PriorityQueue* pq) {
    return (pq->size == MAX);
}

// Enqueue (Insert element at end)
void enqueue(PriorityQueue* pq, int value) {
    if (isFull(pq)) {
        printf("Priority Queue Overflow!\n");
        return;
    }
    pq->arr[pq->size++] = value;
}

// Dequeue (Remove element with highest priority â†’ min here)
int dequeue(PriorityQueue* pq) {
    if (isEmpty(pq)) {
        printf("Priority Queue Underflow!\n");
        return -1;
    }

    // Find minimum element index
    int minIndex = 0;
    for (int i = 1; i < pq->size; i++) {
        if (pq->arr[i] < pq->arr[minIndex]) {
            minIndex = i;
        }
    }

    int minValue = pq->arr[minIndex];

    // Shift elements left
    for (int i = minIndex; i < pq->size - 1; i++) {
        pq->arr[i] = pq->arr[i + 1];
    }

    pq->size--;

    return minValue;
}

// Peek (view highest priority element without removing)
int peek(PriorityQueue* pq) {
    if (isEmpty(pq)) {
        printf("Priority Queue is empty!\n");
        return -1;
    }

    int minIndex = 0;
    for (int i = 1; i < pq->size; i++) {
        if (pq->arr[i] < pq->arr[minIndex]) {
            minIndex = i;
        }
    }

    return pq->arr[minIndex];
}

// Display Queue
void display(PriorityQueue* pq) {
    if (isEmpty(pq)) {
        printf("Priority Queue is empty.\n");
        return;
    }

    printf("Priority Queue: ");
    for (int i = 0; i < pq->size; i++) {
        printf("%d ", pq->arr[i]);
    }
    printf("\n");
}

int main() {
    PriorityQueue pq;
    init(&pq);

    enqueue(&pq, 30);
    enqueue(&pq, 10);
    enqueue(&pq, 40);
    enqueue(&pq, 20);

    display(&pq);

    printf("Peek (highest priority = min): %d\n", peek(&pq));

    printf("Dequeued: %d\n", dequeue(&pq));
    display(&pq);

    printf("Dequeued: %d\n", dequeue(&pq));
    display(&pq);

    return 0;
}


```
