# Develop a program to evaluate a postfix expression using a Stack.

```c

#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <math.h>

typedef struct {
    int* arr;
    int top;
    int capacity;
} Stack;

Stack* createStack(int capacity) {
    Stack* stack = (Stack*)malloc(sizeof(Stack));
    stack->capacity = capacity;
    stack->top = -1;
    stack->arr = (int*)malloc(stack->capacity * sizeof(int));
    return stack;
}

// Check if empty
int isEmpty(Stack* stack) {
    return stack->top == -1;
}

// Push
void push(Stack* stack, int value) {
    stack->arr[++stack->top] = value;
}

// Pop
int pop(Stack* stack) {
    if (isEmpty(stack)) {
        printf("Error: Stack underflow\n");
        exit(1);
    }
    return stack->arr[stack->top--];
}

// Postfix Evaluation 
int evaluatePostfix(char* postfix) {
    Stack* stack = createStack(100);

    for (int i = 0; postfix[i] != '\0'; i++) {
        char ch = postfix[i];

        // Operand (digit)
        if (isdigit(ch)) {
            push(stack, ch - '0');  // convert char → int
        }
        // Operator
        else {
            int val2 = pop(stack);
            int val1 = pop(stack);
            int result;

            switch (ch) {
                case '+': result = val1 + val2; break;
                case '-': result = val1 - val2; break;
                case '*': result = val1 * val2; break;
                case '/': result = val1 / val2; break;
                case '^': result = (int)pow(val1, val2); break;
                default:
                    printf("Error: Invalid operator %c\n", ch);
                    exit(1);
            }
            push(stack, result);
        }
    }

    int finalResult = pop(stack);
    free(stack->arr);
    free(stack);
    return finalResult;
}

int main() {
    char postfix[] = "231*+9-";  // equivalent to: 2 + (3 * 1) - 9

    int result = evaluatePostfix(postfix);

    printf("Postfix Expression: %s\n", postfix);
    printf("Evaluated Result  : %d\n", result);

    return 0;
}


```

# output:

```c

Postfix Expression: 231*+9-
Evaluated Result  : -4

```

## Explanation

- Traverse postfix string:

- If digit → push into stack.

- If operator → pop two elements, apply operator, push result back.

- At the end → stack has one final result.

## Example:

```c

Postfix: 231*+9-
Step1: push(2), push(3), push(1)
Stack: [2, 3, 1]
Step2: apply '*' → 3*1 = 3, push(3)
Stack: [2, 3]
Step3: apply '+' → 2+3 = 5, push(5)
Stack: [5]
Step4: push(9)
Stack: [5, 9]
Step5: apply '-' → 5-9 = -4
Result = -4

```
