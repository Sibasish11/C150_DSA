# Write a program to check if a given string of parentheses is balanced using a Stack.

```c

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char* arr;
    int top;
    int capacity;
} Stack;

// Initialize stack
Stack* createStack(int capacity) {
    Stack* stack = (Stack*)malloc(sizeof(Stack));
    stack->capacity = capacity;
    stack->top = -1;
    stack->arr = (char*)malloc(stack->capacity * sizeof(char));
    return stack;
}

// Check if stack is empty
int isEmpty(Stack* stack) {
    return stack->top == -1;
}

// Push character to stack
void push(Stack* stack, char ch) {
    stack->arr[++stack->top] = ch;
}

// Pop character from stack
char pop(Stack* stack) {
    if (isEmpty(stack)) return '\0';
    return stack->arr[stack->top--];
}

// Peek top element
char peek(Stack* stack) {
    if (isEmpty(stack)) return '\0';
    return stack->arr[stack->top];
}

int isBalanced(char* expr) {
    int n = strlen(expr);
    Stack* stack = createStack(n);

    for (int i = 0; i < n; i++) {
        char ch = expr[i];

        // If opening bracket, push
        if (ch == '(' || ch == '{' || ch == '[') {
            push(stack, ch);
        }
        // If closing bracket, check stack top
        else if (ch == ')' || ch == '}' || ch == ']') {
            if (isEmpty(stack)) return 0;

            char top = pop(stack);

            if ((ch == ')' && top != '(') ||
                (ch == '}' && top != '{') ||
                (ch == ']' && top != '[')) {
                return 0; // mismatch
            }
        }
    }

    int balanced = isEmpty(stack);
    free(stack->arr);
    free(stack);
    return balanced;
}

int main() {
    char expr1[] = "{[()]}";
    char expr2[] = "{[(])}";
    char expr3[] = "((()))";

    printf("%s -> %s\n", expr1, isBalanced(expr1) ? "Balanced" : "Not Balanced");
    printf("%s -> %s\n", expr2, isBalanced(expr2) ? "Balanced" : "Not Balanced");
    printf("%s -> %s\n", expr3, isBalanced(expr3) ? "Balanced" : "Not Balanced");

    return 0;
}

```
