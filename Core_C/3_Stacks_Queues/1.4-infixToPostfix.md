# Implement a program to convert an infix expression to a postfix expression using a Stack.

```c

#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <string.h>

typedef struct {
    char* arr;
    int top;
    int capacity;
} Stack;

// Create stack
Stack* createStack(int capacity) {
    Stack* stack = (Stack*)malloc(sizeof(Stack));
    stack->capacity = capacity;
    stack->top = -1;
    stack->arr = (char*)malloc(stack->capacity * sizeof(char));
    return stack;
}

// Check if empty
int isEmpty(Stack* stack) {
    return stack->top == -1;
}

// Push element
void push(Stack* stack, char ch) {
    stack->arr[++stack->top] = ch;
}

// Pop element
char pop(Stack* stack) {
    if (isEmpty(stack)) return '\0';
    return stack->arr[stack->top--];
}

// Peek element
char peek(Stack* stack) {
    if (isEmpty(stack)) return '\0';
    return stack->arr[stack->top];
}

// Operator Utility Functions 
int precedence(char op) {
    switch (op) {
        case '+':
        case '-': return 1;
        case '*':
        case '/': return 2;
        case '^': return 3;
    }
    return 0;
}

int isOperator(char ch) {
    return (ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '^');
}

// Infix → Postfix Conversion 
void infixToPostfix(char* infix, char* postfix) {
    Stack* stack = createStack(strlen(infix));
    int i, k = 0;

    for (i = 0; infix[i] != '\0'; i++) {
        char ch = infix[i];

        // Operand → Add to postfix
        if (isalnum(ch)) {
            postfix[k++] = ch;
        }
        // Opening parenthesis → Push
        else if (ch == '(') {
            push(stack, ch);
        }
        // Closing parenthesis → Pop until '('
        else if (ch == ')') {
            while (!isEmpty(stack) && peek(stack) != '(') {
                postfix[k++] = pop(stack);
            }
            pop(stack); // remove '('
        }
        // Operator
        else if (isOperator(ch)) {
            while (!isEmpty(stack) && precedence(peek(stack)) >= precedence(ch)) {
                postfix[k++] = pop(stack);
            }
            push(stack, ch);
        }
    }

    // Pop remaining operators
    while (!isEmpty(stack)) {
        postfix[k++] = pop(stack);
    }

    postfix[k] = '\0';

    free(stack->arr);
    free(stack);
}

int main() {
    char infix[] = "A+(B*C-(D/E^F)*G)*H";
    char postfix[100];

    infixToPostfix(infix, postfix);

    printf("Infix Expression  : %s\n", infix);
    printf("Postfix Expression: %s\n", postfix);

    return 0;
}

```

Postfix Expression: ABC*DEF^/G*-H*
