# Implement a Deque (Double-Ended Queue) data structure using a Doubly Linked List. Include addFront, addRear, removeFront, removeRear operations.

```c

#include <stdio.h>
#include <stdlib.h>

// Node structure for Doubly Linked List
typedef struct Node {
    int data;
    struct Node* prev;
    struct Node* next;
} Node;

// Deque structure
typedef struct {
    Node* front;
    Node* rear;
} Deque;

// Create a new node
Node* createNode(int value) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    newNode->data = value;
    newNode->prev = NULL;
    newNode->next = NULL;
    return newNode;
}

// Initialize Deque
void initDeque(Deque* dq) {
    dq->front = dq->rear = NULL;
}

// Check if empty
int isEmpty(Deque* dq) {
    return (dq->front == NULL);
}

// Add element at front
void addFront(Deque* dq, int value) {
    Node* newNode = createNode(value);
    if (isEmpty(dq)) {
        dq->front = dq->rear = newNode;
    } else {
        newNode->next = dq->front;
        dq->front->prev = newNode;
        dq->front = newNode;
    }
    printf("%d added at front.\n", value);
}

// Add element at rear
void addRear(Deque* dq, int value) {
    Node* newNode = createNode(value);
    if (isEmpty(dq)) {
        dq->front = dq->rear = newNode;
    } else {
        newNode->prev = dq->rear;
        dq->rear->next = newNode;
        dq->rear = newNode;
    }
    printf("%d added at rear.\n", value);
}

// Remove element from front
int removeFront(Deque* dq) {
    if (isEmpty(dq)) {
        printf("Deque Underflow! Nothing to remove.\n");
        return -1;
    }
    Node* temp = dq->front;
    int value = temp->data;

    dq->front = dq->front->next;
    if (dq->front != NULL)
        dq->front->prev = NULL;
    else
        dq->rear = NULL; // If deque becomes empty

    free(temp);
    return value;
}

// Remove element from rear
int removeRear(Deque* dq) {
    if (isEmpty(dq)) {
        printf("Deque Underflow! Nothing to remove.\n");
        return -1;
    }
    Node* temp = dq->rear;
    int value = temp->data;

    dq->rear = dq->rear->prev;
    if (dq->rear != NULL)
        dq->rear->next = NULL;
    else
        dq->front = NULL; // If deque becomes empty

    free(temp);
    return value;
}

// Display Deque
void display(Deque* dq) {
    if (isEmpty(dq)) {
        printf("Deque is empty.\n");
        return;
    }
    Node* temp = dq->front;
    printf("Deque: ");
    while (temp != NULL) {
        printf("%d ", temp->data);
        temp = temp->next;
    }
    printf("\n");
}

int main() {
    Deque dq;
    initDeque(&dq);

    addRear(&dq, 10);
    addRear(&dq, 20);
    addFront(&dq, 5);
    addFront(&dq, 2);
    display(&dq);

    printf("Removed from front: %d\n", removeFront(&dq));
    display(&dq);

    printf("Removed from rear: %d\n", removeRear(&dq));
    display(&dq);

    return 0;
}


```
