# Develop a program for browser back/forward functionality using two stacks.
  
## Idea:
  - visitPage(x): Push currentPage to backStack, clear forwardStack, set currentPage = x
  - back(): Push currentPage to forwardStack, pop from backStack into currentPage.
  - forward(): Push currentPage to backStack, pop from forwardStack into currentPage.

```c

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int top;
    int capacity;
    char** arr; // store page names (strings)
} Stack;

// Create stack
Stack* createStack(int capacity) {
    Stack* stack = (Stack*)malloc(sizeof(Stack));
    stack->capacity = capacity;
    stack->top = -1;
    stack->arr = (char**)malloc(capacity * sizeof(char*));
    return stack;
}

// Check empty
int isEmpty(Stack* stack) {
    return stack->top == -1;
}

// Push
void push(Stack* stack, const char* page) {
    if (stack->top == stack->capacity - 1) {
        printf("Stack Overflow!\n");
        return;
    }
    stack->arr[++stack->top] = strdup(page);
}

// Pop
char* pop(Stack* stack) {
    if (isEmpty(stack)) return NULL;
    return stack->arr[stack->top--];
}

// Peek
char* peek(Stack* stack) {
    if (isEmpty(stack)) return NULL;
    return stack->arr[stack->top];
}

// Clear stack
void clearStack(Stack* stack) {
    stack->top = -1;
}

// Browser Functionality 
typedef struct {
    char* currentPage;
    Stack* backStack;
    Stack* forwardStack;
} Browser;

// Create Browser
Browser* createBrowser(int capacity) {
    Browser* browser = (Browser*)malloc(sizeof(Browser));
    browser->currentPage = NULL;
    browser->backStack = createStack(capacity);
    browser->forwardStack = createStack(capacity);
    return browser;
}

// Visit a new page
void visitPage(Browser* browser, const char* url) {
    if (browser->currentPage != NULL) {
        push(browser->backStack, browser->currentPage);
    }
    browser->currentPage = strdup(url);
    clearStack(browser->forwardStack); // forward history cleared
    printf("Visited: %s\n", browser->currentPage);
}

// Go Back
void goBack(Browser* browser) {
    if (isEmpty(browser->backStack)) {
        printf("No pages in Back history!\n");
        return;
    }
    push(browser->forwardStack, browser->currentPage);
    browser->currentPage = pop(browser->backStack);
    printf("Back to: %s\n", browser->currentPage);
}

// Go Forward
void goForward(Browser* browser) {
    if (isEmpty(browser->forwardStack)) {
        printf("No pages in Forward history!\n");
        return;
    }
    push(browser->backStack, browser->currentPage);
    browser->currentPage = pop(browser->forwardStack);
    printf("Forward to: %s\n", browser->currentPage);
}

int main() {
    Browser* browser = createBrowser(10);

    visitPage(browser, "google.com");
    visitPage(browser, "github.com");
    visitPage(browser, "stackoverflow.com");

    goBack(browser);
    goBack(browser);
    goForward(browser);

    visitPage(browser, "openai.com"); // clears forward history
    goBack(browser);

    return 0;
}


```
