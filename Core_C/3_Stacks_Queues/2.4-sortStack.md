# Implement a program to sort a stack using another temporary stack.

```c

#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int top;
    int capacity;
    int* arr;
} Stack;

// Create stack
Stack* createStack(int capacity) {
    Stack* stack = (Stack*)malloc(sizeof(Stack));
    stack->capacity = capacity;
    stack->top = -1;
    stack->arr = (int*)malloc(capacity * sizeof(int));
    return stack;
}

// Check empty
int isEmpty(Stack* stack) {
    return stack->top == -1;
}

// Push
void push(Stack* stack, int value) {
    if (stack->top == stack->capacity - 1) {
        printf("Stack Overflow!\n");
        return;
    }
    stack->arr[++stack->top] = value;
}

// Pop
int pop(Stack* stack) {
    if (isEmpty(stack)) {
        printf("Stack Underflow!\n");
        return -1;
    }
    return stack->arr[stack->top--];
}

// Peek
int peek(Stack* stack) {
    if (isEmpty(stack)) return -1;
    return stack->arr[stack->top];
}

// Print Stack (top to bottom)
void printStack(Stack* stack) {
    for (int i = stack->top; i >= 0; i--) {
        printf("%d ", stack->arr[i]);
    }
    printf("\n");
}

// Sorting Function 
void sortStack(Stack* input) {
    Stack* tempStack = createStack(input->capacity);

    while (!isEmpty(input)) {
        // Pop element from input
        int temp = pop(input);

        // While temporary stack not empty and top > temp
        while (!isEmpty(tempStack) && peek(tempStack) > temp) {
            push(input, pop(tempStack));
        }

        // Place temp at correct position in tempStack
        push(tempStack, temp);
    }

    // Copy back to input stack (so it remains sorted)
    while (!isEmpty(tempStack)) {
        push(input, pop(tempStack));
    }

    free(tempStack->arr);
    free(tempStack);
}

int main() {
    Stack* stack = createStack(10);

    // Push unsorted values
    push(stack, 34);
    push(stack, 3);
    push(stack, 31);
    push(stack, 98);
    push(stack, 92);
    push(stack, 23);

    printf("Original Stack (top to bottom):\n");
    printStack(stack);

    sortStack(stack);

    printf("Sorted Stack (top to bottom):\n");
    printStack(stack);

    return 0;
}


```
