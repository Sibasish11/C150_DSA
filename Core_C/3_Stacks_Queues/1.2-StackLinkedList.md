# Implement a Stack data structure using a Linked List. Include push, pop, peek, and isEmpty operations.

```c

#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

typedef struct {
    Node* top;  // Pointer to the top node
} Stack;

// Initialize stack
void initStack(Stack* s) {
    s->top = NULL;
}

// Check if stack is empty
int isEmpty(Stack* s) {
    return s->top == NULL;
}

// Push an element onto the stack
void push(Stack* s, int value) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    if (!newNode) {
        printf("Error: Heap overflow\n");
        return;
    }
    newNode->data = value;
    newNode->next = s->top; // New node points to old top
    s->top = newNode;       // Top becomes new node
    printf("%d pushed to stack\n", value);
}

// Pop an element from the stack
int pop(Stack* s) {
    if (isEmpty(s)) {
        printf("Error: Stack Underflow\n");
        return -1;
    }
    Node* temp = s->top;
    int poppedValue = temp->data;
    s->top = temp->next; // Move top to next node
    free(temp);
    return poppedValue;
}

// Peek at the top element without removing it
int peek(Stack* s) {
    if (isEmpty(s)) {
        printf("Error: Stack is empty\n");
        return -1;
    }
    return s->top->data;
}

int main() {
    Stack s;
    initStack(&s);

    push(&s, 10);
    push(&s, 20);
    push(&s, 30);

    printf("Top element is %d\n", peek(&s));

    printf("Popped: %d\n", pop(&s));
    printf("Popped: %d\n", pop(&s));

    if (isEmpty(&s)) {
        printf("Stack is empty\n");
    } else {
        printf("Stack is not empty\n");
    }

    return 0;
}

```

## Output:

20 pushed to stack
30 pushed to stack
Top element is 30
Popped: 30
Popped: 20
Stack is not empty
