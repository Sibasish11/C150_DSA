# Develop a program to simulate a simple task scheduler using a Queue.

```c

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Task {
    int id;
    char name[50];
    struct Task* next;
} Task;

typedef struct {
    Task* front;
    Task* rear;
} Queue;

// Initialize queue
void initQueue(Queue* q) {
    q->front = q->rear = NULL;
}

// Check if queue is empty
int isEmpty(Queue* q) {
    return (q->front == NULL);
}

// Enqueue a new task
void enqueue(Queue* q, int id, char* name) {
    Task* newTask = (Task*)malloc(sizeof(Task));
    newTask->id = id;
    strcpy(newTask->name, name);
    newTask->next = NULL;

    if (q->rear == NULL) {
        q->front = q->rear = newTask;
    } else {
        q->rear->next = newTask;
        q->rear = newTask;
    }
    printf("Task Added: ID=%d, Name=%s\n", id, name);
}

// Dequeue a task (execute it)
void dequeue(Queue* q) {
    if (isEmpty(q)) {
        printf("No tasks to execute.\n");
        return;
    }
    Task* temp = q->front;
    printf("Executing Task: ID=%d, Name=%s\n", temp->id, temp->name);

    q->front = q->front->next;
    if (q->front == NULL)
        q->rear = NULL;

    free(temp);
}

// Display all tasks in queue
void display(Queue* q) {
    if (isEmpty(q)) {
        printf("No pending tasks.\n");
        return;
    }
    printf("Pending Tasks:\n");
    Task* current = q->front;
    while (current != NULL) {
        printf("  [ID=%d, Name=%s]\n", current->id, current->name);
        current = current->next;
    }
}

int main() {
    Queue scheduler;
    initQueue(&scheduler);

    // Adding tasks
    enqueue(&scheduler, 1, "Backup Files");
    enqueue(&scheduler, 2, "Update Software");
    enqueue(&scheduler, 3, "Run Virus Scan");

    printf("\n");
    display(&scheduler);

    printf("\n-- Executing Tasks --\n");
    dequeue(&scheduler);
    dequeue(&scheduler);

    printf("\n");
    display(&scheduler);

    printf("\n-- Executing Remaining Tasks --\n");
    dequeue(&scheduler);
    dequeue(&scheduler);  // test underflow

    return 0;
}


```
